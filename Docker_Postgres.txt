Run and configure Postgres in docker 

1. docker pull postgres 
2. docker run -d -p 5432:5432 --name container_name -e POSTGRES_PASSWORD = password image_name
3. docker exec -it container_name bash 

// Prepare database:
1. create database database_name;
2. create user username with password 'password';
3. grant all privileges on database name_db to username;

psql -U username -d database_name
